<script>
    // Get the elements
    const openNavlinks = document.getElementById('openNavlinks');
    const closeNavlinks = document.getElementById('closeNavlinks');
    const navlinks = document.querySelector('.nav-links');

    // Event listener to open nav links
    openNavlinks.addEventListener('click', () => {
        navlinks.classList.add('active'); // Activate the .navlinks
        openNavlinks.style.display = 'none'; // Hide the open (bars) icon
        closeNavlinks.style.display = 'flex'; // Show the close (x) icon
    });

    // Event listener to close nav links
    closeNavlinks.addEventListener('click', () => {
        navlinks.classList.remove('active'); // Deactivate the .navlinks
        closeNavlinks.style.display = 'none'; // Hide the close (x) icon
        openNavlinks.style.display = 'flex'; // Show the open (bars) icon
    });

</script>
